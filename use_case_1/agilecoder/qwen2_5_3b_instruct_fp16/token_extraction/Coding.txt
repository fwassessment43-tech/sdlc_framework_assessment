class Game:
    def __init__(self, snake, food, scoreboard):
        self.snake = snake
        self.food = food
        self.scoreboard = scoreboard

    def update(self):
        # Move the snake and handle collisions
        self.snake.move()

        # Check for collision with the board boundaries or itself.
        if self.board_collision() or self.snake_head_collision():
            print("Game Over")
            return

        # Generate new food randomly on the board, ensuring it's not on the snake.
        while self.food.position in [segment.position for segment in self.snake.segments]:
            self.generate_food()

        self.scoreboard.draw_score(self.screen)

    def generate_food(self):
        possible_positions = [(x, y) for x in range(10) for y in range(10)]
        position = tuple(random.choice(possible_positions))
        while position in [segment.position for segment in self.snake.segments]:
            position = tuple(random.choice(possible_positions))

        self.food.position = position

    def board_collision(self):
        # Check boundaries
        head_position = self.snake.segments[0].position
        if (head_position[0] < 0 or head_position[0] >= 10 or
            head_position[1] < 0 or head_position[1] >= 10):
            return True
        return False

    def snake_head_collision(self):
        # Check self-collision
        for segment in self.snake.segments:
            if segment.position == self.snake.segments[0].position:
                return True
        return False

'''
Snake initialization setup.
Starts a snake on the game board with initial length and direction.
The snake moves right by default.
'''
from typing import List, Tuple
import random

class Snake:
    def __init__(self):
        self.length = 1
        # Initial position at the center of the grid
        self.body: List[Tuple[int]] = [(GRID_SIZE // 2 - self.length // 2 + i, GRID_SIZE // 2) for i in range(self.length)]
        self.direction = 'right'

    def move_snake(self):
        new_head = self.get_new_head()
        if not self.check_collision(new_head):
            self.body.insert(0, new_head)
            self.body.pop()

    def get_new_head(self) -> Tuple[int]:
        head_x, head_y = self.body[0]
        if self.direction == 'right': return (head_x + 1, head_y)
        elif self.direction == 'left': return (head_x - 1, head_y)
        elif self.direction == 'up': return (head_x, head_y - 1)
        else: return (head_x, head_y + 1)

    def check_collision(self, position: Tuple[int]) -> bool:
        """Check if the snake collides with its own body or the boundaries."""
        return position in self.body[:-1] or any([pos[0] < 0 or pos[0] >= GRID_SIZE or pos[1] < 0 or pos[1] >= GRID_SIZE for pos in self.body])

snake = Snake()

        new_head = (head_x, head_y + 1)
    elif direction == 'LEFT':
        new_head = (head_x - 1, head_y)
    elif direction == 'RIGHT':
        new_head = (head_x + 1, head_y)
    snake.insert(0, new_head)

    return False

    size_x, size_y = board_size

class Grid:
    def __init__(self, width=10, height=10):
        self.width = width
        self.height = height

    def draw(self):
        for y in range(self.height):
            if y == 0 or y == self.height - 1:
                print('#' * (self.width + 2))
            else:
                row = '#' + ' ' * self.width + '#'
                print(row)
        print('#' * (self.width + 2))

class Snake:
    def __init__(self, grid):
        self.grid = grid
        self.body = [(3, 5), (4, 5)]  # Initial snake position and length.
        self.direction = 'right'

    def move(self):
        head_x, head_y = self.body[0]
        if self.direction == 'up':
            new_head = (head_x, head_y - 1)
        elif self.direction == 'down':
            new_head = (head_x, head_y + 1)
        elif self.direction == 'left':
            new_head = (head_x - 1, head_y)
        else:  # Direction is 'right'
            new_head = (head_x + 1, head_y)

        if not (0 <= new_head[0] < self.grid.width and
                0 <= new_head[1] < self.grid.height):
            return False

        self.body.insert(0, new_head)
        if len(self.body) > self.grid.width * self.grid.height:
            del self.body[-1]

        return True

    def grow(self):
        tail_x, tail_y = self.body[-1]
        if self.direction == 'up':
            new_tail = (tail_x, tail_y + 1)
        elif self.direction == 'down':
            new_tail = (tail_x, tail_y - 1)
        elif self.direction == 'left':
            new_tail = (tail_x + 1, tail_y)
        else:  # Direction is 'right'
            new_tail = (tail_x - 1, tail_y)

        if not (0 <= new_tail[0] < self.grid.width and
                0 <= new_tail[1] < self.grid.height):
            return False

        self.body.append(new_tail)
        return True

import random
from enum import Enum

class Direction(Enum):
    RIGHT = '>'
    LEFT = '<'
    UP = '^'
    DOWN = 'v'

def draw_board(board, snake_position, food_position, direction):
    print('+' + ('-' * 10) + '+')
    for row in board:
        line = '|'
        for cell in row:
            if (cell[0], cell[1]) == snake_position:
                line += Direction(direction).value
            elif (cell[0], cell[1]) == food_position:
                line += 'F'
            else:
                line += '#'
        print(line + '|')
    print('+' + ('-' * 10) + '+')


def generate_food(snake_positions):
    available_cells = [(x, y) for x in range(10) for y in range(10)]
    for pos in snake_positions:
        if pos in available_cells:
            available_cells.remove(pos)
    food_position = random.choice(available_cells)
    return food_position

    board = [['#' for _ in range(10)] for _ in range(10)]

        head_x, head_y = initial_snake[0]
        if direction == 'RIGHT':
            new_head = (head_x + 1, head_y)
        elif direction == 'LEFT':
            new_head = (head_x - 1, head_y)
        elif direction == 'UP':
            new_head = (head_x, head_y - 1)
        else:
            new_head = (head_x, head_y + 1)

        if new_head not in initial_snake and new_head != food_position:
            del snake_positions[-1]
            snake_positions.insert(0, new_head)

    # Initial setup

    while True:

class GameBoard:
    def __init__(self, width=10, height=10):
        self.width = width
        self.height = height
        self.board = [[' ' for _ in range(width)] for _ in range(height)]
        self.snake = [(random.randint(2, width-3), random.randint(2, height-3))]
        self.food = self.spawn_food()
        self.direction = (0, 1)  # Moving right initially

    def spawn_food(self):
        x, y = random.randint(2, self.width - 3), random.randint(2, self.height - 3)
        while (x, y) in self.snake:
            x, y = random.randint(2, self.width - 3), random.randint(2, self.height - 3)
        return (x, y)

    def move_snake(self):
        head_x, head_y = self.snake[0]
        new_head = ((head_x + self.direction[0]) % self.width,
                    (head_y + self.direction[1]) % self.height)
        if new_head in self.snake:
            raise Exception("Game Over: Snake ate itself!")
        self.snake.insert(0, new_head)

    def draw_board(self):
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) == tuple(self.snake[-1]):
                    print('O', end=' ')
                elif (x, y) == self.food:
                    print('~', end=' ')
                else:
                    print(' ', end=' ')

    def handle_key_press(self, key):
        # Key bindings: w/a/s/d for up/left/down/right

            return (0, -1)

            return (-1, 0)  # Left
        elif key == 'd' and not self.direction[0] == -1:
            return (1, 0)  # Right
        else:
            return self.direction

    def game_over(self):

def handle_movement(board, direction):
    """
    Handles the movement of the snake based on the given direction.

    :param board: The current state of the game board (list).
    :param direction: A string representing the new direction ('left', 'right', 'up', 'down').
    """
    head_x, head_y = next(iter(board[0]))  # Get the top-left corner

    if direction == 'left':
        new_head = (head_x - 1, head_y)
    elif direction == 'right':
        new_head = (head_x + 1, head_y)
    elif direction == 'up':
        new_head = (head_x, head_y - 1)
    else:  # Down
        new_head = (head_x, head_y + 1)

    if not is_valid_move(board, new_head):
        return False

    for body_part in board[0]:
        board[board.index([body_part])].remove(body_part)

    snake_body.append(new_head)

    return True

def is_valid_move(board, new_head):
    """
    Checks if a move to the given head position would be valid.

    :param board: The current state of the game board (list).
    :param new_head: A tuple representing the new head position.
    :return: True if valid, False otherwise.
    """
    x, y = new_head
    return 0 < x < 10 and 0 < y < 10

def detect_collision(board):
    """
    Detects collisions with boundaries or snake body.

    :param board: The current state of the game board (list).
    :return: True if a collision is detected, False otherwise.
    """
    head_x, head_y = next(iter(board[0]))
    tail_x, tail_y = snake_body(board)[-1]

    # Check boundaries
    if not is_valid_move(board, (head_x-1, head_y)):
        return True

    if not is_valid_move(board, (head_x+1, head_y)):
        return True

    if not is_valid_move(board, (head_x, head_y-1)):
        return True

    if not is_valid_move(board, (head_x, head_y+1)):
        return True

    # Check collision with the snake body
    if tail_x != head_x or tail_y != head_y:
        return True

    return False

def detect_food_collision(board):
    """
    Detects collisions between the snake and food.

    :param board: The current state of the game board (list).
    :return: True if a collision is detected, False otherwise.
    """
    x, y = next(iter(board[0]))
    return 'o' in board[x][y:y+1] or 'o' in board[x-1:x]

    Updates the board to reflect the new position of the head after eating food.

    :param board: The current state of the game board (list).

'''
This script creates a snake game using Python's Tkinter library.
The game features a grid-based board, a moving snake, food generation,
collision handling, and score display.
'''
import tkinter as tk
import random

# Constants
GRID_SIZE = 10
BOARD_WIDTH = GRID_SIZE * 20
BOARD_HEIGHT = GRID_SIZE * 20
SNAKE_SPEED = 100

class SnakeGame:
    def __init__(self):
        self.root = tk.Tk()
        self.canvas = tk.Canvas(self.root, width=BOARD_WIDTH, height=BOARD_HEIGHT)
        self.canvas.pack()

        # Initialize snake and food
        self.snake_body = [(20, 20), (40, 20)]
        self.food_pos = self.generate_food_position()
        self.direction = 'RIGHT'

        # Bind arrow keys to direction changes
        self.root.bind('<Left>', lambda event: self.change_direction('LEFT'))
        self.root.bind('<Right>', lambda event: self.change_direction('RIGHT'))
        self.root.bind('<Up>', lambda event: self.change_direction('UP'))
        self.root.bind('<Down>', lambda event: self.change_direction('DOWN'))

        # Start game loop
        self.start_game()

    def start_game(self):
        self.update_snake_position()
        self.canvas.after(SNAKE_SPEED, self.start_game)

    def update_snake_position(self):
        x1, y1 = self.snake_body[0]
        if self.direction == 'LEFT':
            new_head_pos = (x1 - 20, y1)
        elif self.direction == 'RIGHT':
            new_head_pos = (x1 + 20, y1)
        elif self.direction == 'UP':
            new_head_pos = (x1, y1 - 20)
        else:
            new_head_pos = (x1, y1 + 20)

        # Check for collision with boundaries
        if not (0 <= new_head_pos[0] < BOARD_WIDTH and 0 <= new_head_pos[1] < BOARD_HEIGHT):
            self.game_over()
            return

        # Check for snake's own body collision
        if new_head_pos in self.snake_body[1:]:
            self.game_over()
            return

        # Update snake position
        self.snake_body.insert(0, new_head_pos)
        if not (new_head_pos == self.food_pos):
            self.snake_body.pop()

        # Check for food collision and increase score
        if new_head_pos == self.food_pos:
            self.score += 1
            self.food_pos = self.generate_food_position()
        else:
            self.canvas.delete('food')

        # Draw snake and food
        self.draw_snake()
        self.draw_food()

    def generate_food_position(self):
        x = random.randint(1, GRID_SIZE - 2) * 20
        y = random.randint(1, GRID_SIZE - 2) * 20
        return (x, y)

    def draw_snake(self):
        self.canvas.delete('snake')
        for segment in self.snake_body:
            self.canvas.create_rectangle(segment[0], segment[1], segment[0] + 20, segment[1] + 20, fill='green', tags='snake')

        if (food_x, food_y) not in body_segments:

import game_board

class Snake:
    def __init__(self, board_size=10):
        self.board = GameBoard(board_size)
        head_position = (random.randint(2, board_size-3), random.randint(2, board_size-3))
        segments = [(head_position[0], head_position[1] - i) for i in range(3, 0, -1)]
        self.segments = segments
        self.board.grid[segments[0][0]][segments[0][1]] = 'S'

    def move_snake(self, direction):
        # Update snake's position based on the given direction
        if direction == 'UP':
            self.segments = [(segment[0], segment[1] - 1) for segment in self.segments]
        elif direction == 'DOWN':
            self.segments = [(segment[0], segment[1] + 1) for segment in self.segments]
        elif direction == 'LEFT':
            self.segments = [(segment[0] - 1, segment[1]) for segment in self.segments]
        elif direction == 'RIGHT':
            self.segments = [(segment[0] + 1, segment[1]) for segment in self.segments]

    def is_collision(self):
        head = self.segments[0]
        return (head[0] < 2 or head[0] >= len(self.board.grid) - 1 or
                head[1] < 2 or head[1] >= len(self.board.grid[0]) - 1)

    def grow_snake(self):
        food_position = self.board.place_food()
        if food_position == (self.segments[-1][0], self.segments[-1][1]):
            new_segment = self.segments[-1]
            self.segments.append(new_segment)

'''
Snake Game Implementation

This module contains the implementation for a snake game.
It includes functionalities such as snake movement,
food generation, collision detection, and score tracking.

Attributes:
    None

Functions:
    setup_board: Sets up the game board.
    initialize_snake: Initializes the snake on the board.
    handle_key_input: Handles arrow key input for snake movement.
    update_game_state: Updates the state of the game.
    check_collisions: Checks for collisions with boundaries and body.
'''
import random
from collections import deque

def setup_board(width, height):
    """
    Sets up a grid-based game board.

    Args:
        width (int): The width of the game board.
        height (int): The height of the game board.

    Returns:
        tuple: A 2D list representing the game board with obstacles removed for snake movement.
    """
    return [[' ' for _ in range(width)] for _ in range(height)]

def initialize_snake(snake_start_position, direction):
    """
    Initializes the snake on the game board.

    Args:
        snake_start_position (tuple): The starting position of the snake [x, y].
        direction (str): Initial direction the snake is moving ('right', 'left', 'up', 'down').

    Returns:
        list: A deque representing the snake's body with initial length.
    """
    return deque([[snake_start_position[0], snake_start_position[1]]])

def handle_key_input(snake, key):
    """
    Handles arrow key input for snake movement. The direction changes based on the user input.

    Args:
        snake (list): A deque representing the snake's body.
        key (str): Arrow key pressed ('left', 'right', 'up', 'down').

    Returns:
        list: Updated snake with new head position.
    """
    if key == 'left':
        return [[snake[0][0], snake[0][1] - 1]] + snake
    elif key == 'right':
        return [[snake[0][0], snake[0][1] + 1]] + snake
    elif key == 'up':
        return [[snake[0][0] - 1, snake[0][1]]] + snake
    elif key == 'down':
        return [[snake[0][0] + 1, snake[0][1]]] + snake

def update_game_state(snake, food):
    """
    Updates the state of the game by moving the snake and generating new food if necessary.

    Args:
        snake (list): A deque representing the snake's body.
        food (tuple): The position of the current food [x, y].

    Returns:
        list: Updated snake with its head moved to the next position.
        tuple: New position for the food after consumption or generation.
    """
    new_head = [[snake[0][0], snake[0][1]]]
    snake.append(new_head.pop(0))

    if snake[-1] == food:
        return snake, None
    else:
        snake.popleft()
        if not food:
            food = [random.randint(1, width - 2), random.randint(1, height - 2)]
        return snake, food

def check_collisions(snake, board_height, board_width):
    """
    Checks for collisions with boundaries and body.

    Args:
        snake (list): A deque representing the snake's body.
        board_height (int): Height of the game board.
        board_width (int): Width of the game board.

    Returns:
        bool: True if collision is detected, False otherwise.
    """
    head = snake[0]
    return (
        0 <= head[0] < board_width and
        0 <= head[1] < board_height and
        not all(head == segment for segment in snake[1:])
    )

    height = 20

        print('\n'.join([''.join(row) for row in game_board]))

        snake, food = update_game_state(snake, food)
        if check_collisions(snake, height, width):
            break

    print("\nGame Over! You collided with the board or your own body.")

    def get_food_position():
        while True:
            food_x, food_y = random.randint(1, width - 2), random.randint(1, height - 2)

        """Draws the snake on the grid."""
        for pos in body_positions:
            print(f"\u25A0" if (pos.x, pos.y) == head else " ", end="")

        print("\u2B1C" if food_pos != body_positions[0] else "", end="")

def create_snake_game():
    """
    Create a snake game with the specified requirements.

    This function initializes and runs a game where the player controls a snake to navigate through a grid-based board.
    The game ends when the snake collides with itself or the boundaries of the board, and it displays a score for each successful movement.
    """
    import random
    import time

    # Define the dimensions of the game board (10x10)
    BOARD_SIZE = 10
    # Initialize the snake's head position
    HEAD_X, HEAD_Y = BOARD_SIZE // 2 - 1, BOARD_SIZE // 2 - 1
    SNAKE_BODY = [(HEAD_X, HEAD_Y)]
    # Define initial direction of movement (right)
    DIRECTION = (0, 1)

    def draw_board():
        """Draw the game board and snake."""
        print('\n' * 50)  # Clear screen for a clean start
        for y in range(BOARD_SIZE):
            row = []
            for x in range(BOARD_SIZE):
                if (x, y) == HEAD_X, HEAD_Y:
                    row.append('S')
                elif (x, y) in SNAKE_BODY[1:]:
                    row.append('>')
                else:
                    row.append('.')
            print('|' + ''.join(row) + '|')

    def move_snake():
        """Move the snake based on its current direction."""
        global HEAD_X, HEAD_Y
        new_head = (HEAD_X + DIRECTION[0], HEAD_Y + DIRECTION[1])
        if not (0 <= new_head[0] < BOARD_SIZE and 0 <= new_head[1] < BOARD_SIZE):
            game_over()
        elif new_head in SNAKE_BODY:
            game_over()
        else:
            SNAKE_BODY.insert(0, new_head)
            return True
        HEAD_X, HEAD_Y = new_head

    def generate_food():
        """Generate a random food position not on the snake's body."""
        while True:
            x, y = random.randint(0, BOARD_SIZE - 1), random.randint(0, BOARD_SIZE - 1)
            if (x, y) not in SNAKE_BODY:
                return x, y

        print("Game Over!")
        time.sleep(2)
        draw_board()

                del SNAKE_BODY[BOARD_SIZE]

        """Initialize game settings."""

import random

class Board:
    def __init__(self, size=BOARD_SIZE):
        self.size = size
        self.board = [[' ' for _ in range(size[1])] for _ in range(size[0])]
        self.snake = SNAKE_START_POSITION.copy()
        self.food_position = self._generate_food_position()

    def move_snake(self, direction):
        head_x, head_y = self.snake[0]
        if direction == 'UP':
            new_head = (head_x, max(0, head_y - 1))
        elif direction == 'DOWN':
            new_head = (head_x, min(head_y + 1, self.size[1] - 1))
        elif direction == 'LEFT':
            new_head = (max(0, head_x - 1), head_y)
        elif direction == 'RIGHT':
            new_head = (min(head_x + 1, self.size[0] - 1), head_y)

        if tuple(new_head) in self.snake:
            raise CollisionError('Collided with itself')
        if new_head == self.food_position:
            self._generate_food_position()
            return True
        else:
            self.board[head_x][head_y] = ' '
            self.snake.insert(0, new_head)
            return False

    def _generate_food_position(self):
        while True:
            x, y = random.randint(1, self.size[0] - 2), random.randint(1, self.size[1] - 2)
            if (x, y) not in self.snake:
                return (x, y)

    def draw(self):
        for row_idx, row in enumerate(self.board):
            for col_idx, cell in enumerate(row):
                print(cell if (row_idx, col_idx) == (0, 0) else ' ', end='')

class SnakeGame:

        self.grid_size = grid_size
        self.snake_head = [(grid_size // 2, grid_size // 2)]
        self.snake_body = [list(self.snake_head[0])]
        self.direction = 'right'
        self.score = 0

    def draw_board(self):
        print("+" + "-" * (self.grid_size - 1) + "+")
        for y in range(self.grid_size):
            row = "|"
            for x in range(self.grid_size):
                if (x, y) == tuple(self.snake_head[0]):
                    row += "o" + "|"
                elif any((nx, ny) == (x, y) for nx, ny in self.snake_body):
                    row += "O" + "|"

        # Generate food
        if head not in self.snake_body:

'''
This script creates a simple snake game using Python's turtle graphics.
The game board is 20x20, and the snake starts at position (10,10) facing right.
Arrow keys control the movement of the snake, and it grows when it eats food.
Collisions with boundaries or its own body result in game over.
'''

import time
import random

# Setup the screen
import turtle
screen = turtle.Screen()
screen.title("Snake Game")
screen.bgcolor('white')
screen.setup(width=400, height=400)
screen.tracer(0)  # Turns off the screen updates

# Snake and food setup
snake = turtle.Turtle()
snake.shape('square')
snake.color('black')
snake.penup()
snake.goto(-10, 0)

food = turtle.Turtle()
food.shape('circle')
food.color('red')
food.penup()
food.goto(random.randint(-9, 9) * 10, random.randint(-9, 9) * 10)

# Directions
directions = {
    'Up': (0, 10),
    'Down': (0, -10),
    'Left': (-10, 0),
    'Right': (10, 0)
}

score = 0

def move_snake():
    global score
    new_head = snake.pos()
    new_direction = directions[input("Choose your direction [Up/Down/Left/Right]: ").capitalize()]

    if new_direction == directions['Up']:
        snake.goto(new_head[0], new_head[1] + 10)
    elif new_direction == directions['Down']:
        snake.goto(new_head[0], new_head[1] - 10)
    elif new_direction == directions['Left']:
        snake.goto(new_head[0] - 10, new_head[1])
    elif new_direction == directions['Right']:
        snake.goto(new_head[0] + 10, new_head[1])

    if snake.pos() == food.pos():
        score += 1
        # Hide the food and generate a new one
        food.hideturtle()
        food.goto(random.randint(-9, 9) * 10, random.randint(-9, 9) * 10)
        food.showturtle()

# Main game loop
while True:
    move_snake()
    screen.update()
    time.sleep(0.1)

    if snake.pos() == food.pos():
        pass
    elif snake.distance(food) < 10:
        break

snake.hideturtle()
food.hideturtle()

if not 20 > abs(snake.xcor()) >= -19 or not 20 > abs(snake.ycor()) >= -19:
    print("Game Over")
else:
    if snake.distance(food) < 10 and (abs(snake.xcor()) == abs(10) or abs(snake.ycor()) == abs(10)):
        score += 5
    else:
        score -= 5

print(f"Final Score: {score}")

# Exit the game gracefully
turtle.done()
