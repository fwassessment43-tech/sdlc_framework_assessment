```python
import pygame
import sys
import random

# Initialize Pygame
pygame.init()

# Set up some constants
WIDTH, HEIGHT = 800, 600
BLOCK_SIZE = 20

# Set up some colors
WHITE = (255, 255, 255)
RED = (255, 0, 0)
GREEN = (0, 255, 0)

# Set up the display
screen = pygame.display.set_mode((WIDTH, HEIGHT))

# Set up the font
font = pygame.font.Font(None, 36)

# Set up the clock
clock = pygame.time.Clock()

class SnakeGame:
    def __init__(self):
        self.snake = [(200, 200), (220, 200), (240, 200)]
        self.direction = (20, 0)
        self.food = self.generate_food()
        self.score = 0

    def generate_food(self):
        while True:
            x = random.randint(0, WIDTH // BLOCK_SIZE - 1) * BLOCK_SIZE
            y = random.randint(0, HEIGHT // BLOCK_SIZE - 1) * BLOCK_SIZE
            if (x, y) not in self.snake:
                return x, y

    def update(self):
        head_x, head_y = self.snake[-1]
        new_head_x, new_head_y = head_x + self.direction[0], head_y + self.direction[1]

        # Collision detection
        if (new_head_x < 0 or new_head_x >= WIDTH or
            new_head_y < 0 or new_head_y >= HEIGHT or
            (new_head_x, new_head_y) in self.snake[:-1]):
            return False

        self.snake.append((new_head_x, new_head_y))
        if (self.snake[-1] == self.food):
            self.score += 1
            self.food = self.generate_food()
        else:
            self.snake.pop(0)

    def draw(self):
        screen.fill(WHITE)
        for x, y in self.snake:
            pygame.draw.rect(screen, GREEN, (x, y, BLOCK_SIZE, BLOCK_SIZE))
        pygame.draw.rect(screen, RED, (*self.food, BLOCK_SIZE, BLOCK_SIZE))
        text = font.render(f"Score: {self.score}", True, BLACK)
        screen.blit(text, (10, 10))

def main():
    game = SnakeGame()
    while True:
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                pygame.quit()
                sys.exit()
            elif event.type == pygame.KEYDOWN:
                if event.key == pygame.K_UP and game.direction != (0, 20):
                    game.direction = (0, -20)
                elif event.key == pygame.K_DOWN and game.direction != (0, -20):
                    game.direction = (0, 20)
                elif event.key == pygame.K_LEFT and game.direction != (20, 0):
                    game.direction = (-20, 0)
                elif event.key == pygame.K_RIGHT and game.direction != (-20, 0):
                    game.direction = (20, 0)

        if not game.update():
            break

        game.draw()
        pygame.display.flip()
        clock.tick(10)

if __name__ == "__main__":
    main()
```

This code sets up a basic Snake game using Pygame. The game starts with the snake at position `(200, 200)` and the food at a random location on the screen. The player can move the snake using the arrow keys, and the goal is to eat the food without running into the wall or itself.

The game updates the snake's position based on its current direction, checks for collisions with the boundaries or itself, and updates the score accordingly. The game also draws the snake and the food on the screen at their respective positions.
