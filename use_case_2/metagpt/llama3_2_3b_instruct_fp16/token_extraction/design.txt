class TicketManager:
    def __init__(self):
        self.tickets = []
        self.users = []

    def create_ticket(self, title, description, status):
        ticket = Ticket(len(self.tickets) + 1, title, description, status)
        self.tickets.append(ticket)

    def get_ticket(self, id):
        for ticket in self.tickets:
            if ticket.id == id:
                return ticket
        return None

    def update_ticket_status(self, id, status):
        ticket = self.get_ticket(id)
        if ticket:
            ticket.status = status

    def create_user(self, name, email):
        user = User(len(self.users) + 1, name, email)
        self.users.append(user)

    def get_user(self, id):
        for user in self.users:
            if user.id == id:
                return user
        return None

    def assign_ticket_to_staff(self, ticket_id, staff_id):
        ticket = self.get_ticket(ticket_id)
        staff = self.get_user(staff_id)
        if ticket and staff:
