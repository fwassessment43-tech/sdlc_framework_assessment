```python
import tkinter as tk
from tkinter import messagebox
import sqlite3

class TicketManagementSystem:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Ticket Management System")
        self.login_page()

    def login_page(self):
        self.clear_window()
        tk.Label(self.root, text="Login Page").pack()
        tk.Entry(self.root, placeholder="Username").pack()
        tk.Entry(self.root, placeholder="Password").pack()
        tk.Button(self.root, text="Login", command=self.ticket_management_system).pack()

    def ticket_management_system(self):
        self.clear_window()
        tk.Label(self.root, text="Ticket Management System").pack()
        tk.Entry(self.root, placeholder="New Ticket Description").pack()
        tk.Button(self.root, text="Insert New Ticket", command=lambda: messagebox.showinfo("New Ticket", "Ticket inserted successfully")).pack()

    def clear_window(self):
        for widget in self.root.winfo_children():
            widget.destroy()

class DatabaseImplementation:
    def __init__(self):
        self.conn = sqlite3.connect("tickets.db")
        self.cursor = self.conn.cursor()
        self.create_table()

    def create_table(self):
        self.cursor.execute("""
            CREATE TABLE IF NOT EXISTS tickets (
                id INTEGER PRIMARY KEY,
                description TEXT,
                status TEXT,
                category TEXT,
                opening_date DATE,
                last_modification_date DATE,
                closing_date DATE
            )
        """)
        self.conn.commit()

    def insert_data(self, description, status, category, opening_date, last_modification_date, closing_date):
        self.cursor.execute("""
            INSERT INTO tickets (description, status, category, opening_date, last_modification_date, closing_date)
            VALUES (?, ?, ?, ?, ?, ?)
        """, (description, status, category, opening_date, last_modification_date, closing_date))
        self.conn.commit()

    def close_connection(self):
        self.conn.close()

if __name__ == "__main__":
    db = DatabaseImplementation()
    ticket_management_system = TicketManagementSystem()
    ticket_management_system.root.mainloop()
```
